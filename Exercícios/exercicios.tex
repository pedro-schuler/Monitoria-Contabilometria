\section*{Instructions}
Each exercise begins on a new page. Simple linear regressions are solved using the closed-form formulas:
\[
	\hat\beta_1 = \frac{\sum (x_i-\bar x)(y_i-\bar y)}{\sum (x_i-\bar x)^2},\qquad
	\hat\beta_0 = \bar y - \hat\beta_1\bar x
\]
Variance–covariance for the simple model (intercept and slope) is shown using:
\[
	\widehat{\mathrm{Var}}(\hat\beta)=\hat\sigma^2
	\begin{bmatrix}
		\frac{1}{n} + \frac{\bar x^2}{S_{xx}} & -\frac{\bar x}{S_{xx}} \\[4pt]
		-\frac{\bar x}{S_{xx}}                & \frac{1}{S_{xx}}
	\end{bmatrix},\quad S_{xx}=\sum (x_i-\bar x)^2.
\]

\subsection*{Exercise 1 -- Study Hours and Exam Score}
\textbf{Data (classroom context):} Students recorded weekly study hours and their exam score.

\begin{center}
	\begin{tabular}{ccc}
		\toprule
		Obs & Study Hours ($x$) & Exam Score ($y$) \\
		\midrule
		1   & 2                 & 55               \\
		2   & 3                 & 65               \\
		3   & 4                 & 72               \\
		4   & 5                 & 80               \\
		5   & 6                 & 85               \\
		\bottomrule
	\end{tabular}
\end{center}

\textbf{Solution (step-by-step):}
\begin{enumerate}[leftmargin=*]
	\item Compute sample size, sums and means:
	      \begin{align*}
		      n          & = 5,          \\
		      \sum_i x_i & = 20.000000,  \\
		      \sum_i y_i & = 357.000000, \\
		      \bar x     & = 4.000000,   \\
		      \bar y     & = 71.400000.
	      \end{align*}

	\item Compute $S_{xx}$ and $S_{xy}$:
	      \begin{align*}
		      S_{xx} & = \sum_i (x_i-\bar x)^2 = 10.000000,           \\
		      S_{xy} & = \sum_i (x_i-\bar x)(y_i-\bar y) = 75.000000.
	      \end{align*}

	\item Compute OLS coefficients:
	      \begin{align*}
		      \hat\beta_1 & = \frac{S_{xy}}{S_{xx}} = \frac{75.000000}{10.000000} = 7.500000,                 \\[4pt]
		      \hat\beta_0 & = \bar y - \hat\beta_1 \bar x = 71.400000 - 7.500000 \times 4.000000 = 41.400000.
	      \end{align*}

	\item Fitted values and residuals:
	      \begin{align*}
		      \hat y_i & = [56.400000,\;63.900000,\;71.400000,\;78.900000,\;86.400000],               \\
		      u_i      & = y_i - \hat y_i = [-1.400000,\;1.100000,\;0.600000,\;1.100000,\;-1.400000].
	      \end{align*}

	\item SSE and variance estimate:
	      \begin{align*}
		      SSE          & = \sum_i u_i^2 = 6.700000,                         \\
		      \hat\sigma^2 & = \frac{SSE}{n-2} = \frac{6.700000}{3} = 2.233333.
	      \end{align*}

	\item Variance–covariance matrix of $\hat\beta=(\hat\beta_0,\hat\beta_1)'$:
	      \[
		      \widehat{\mathrm{Var}}(\hat\beta)
		      = \hat\sigma^2
		      \begin{bmatrix}
			      \dfrac{1}{n} + \dfrac{\bar x^2}{S_{xx}} & -\dfrac{\bar x}{S_{xx}} \\[6pt]
			      -\dfrac{\bar x}{S_{xx}}                 & \dfrac{1}{S_{xx}}
		      \end{bmatrix}
		      =
		      \begin{bmatrix}
			      4.020000  & -0.893333 \\[4pt]
			      -0.893333 & 0.223333
		      \end{bmatrix}.
	      \]

	\item Interpretation:
	      \begin{enumerate}
		      \item The slope $\hat\beta_1=7.500000$ means: on average, an additional hour of study is associated with a \emph{7.50 point} increase in exam score (ceteris paribus).
		      \item The intercept $\hat\beta_0=41.400000$ is the predicted exam score when study hours $=0$ (interpret with caution if $x=0$ is outside the sample range).
	      \end{enumerate}
\end{enumerate}
